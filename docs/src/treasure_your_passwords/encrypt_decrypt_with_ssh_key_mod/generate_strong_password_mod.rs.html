3//! Generate strong password.
4//!
5//! Starts with a human readable password.  
6//! It hashes it to 32 bytes, just to make it a little more complex.  
7//! Sign it with the ssh private key, that only the owner can user.  
8//! This could be reversed by using the public key. Because of that:  
9//! Hash it into a 32 bytes. This is a one way encryption  
10//! Finally encode it into a ascii7 string with lowercase, uppercase, numerals and few symbols.

18/// Generate strong password.
19///
20/// Starts with a human readable password.  
21/// It hashes it to 32 bytes, just to make it a little more complex.  
22/// Sign it with the ssh private key, that only the owner can user.  
23/// This could be reversed by using the public key. Because of that:  
24/// Hash it into a 32 bytes. This is a one way encryption  
25/// Finally encode it into a ascii7 string with lowercase, uppercase, numerals and few symbols.  
26pub(crate) fn generate_strong_password() -> anyhow::Result<String> {
27    println!("  {YELLOW}Check if the ssh private key exists.{RESET}");
28    let private_key_file_name = crate::TREASURE_CONFIG.get().unwrap().treasure_private_key_file_name.to_string();
29    let private_key_path_struct = ende::PathStructInSshFolder::new(private_key_file_name.clone()).log()?;
30    if !private_key_path_struct.exists() {
31        eprintln!("{RED}Error: Private key {private_key_path_struct} does not exist.{RESET}");
32        println!("  {YELLOW}Create the private key in bash terminal:{RESET}");
33        println!(r#"{GREEN}ssh-keygen -t ed25519 -f {private_key_path_struct} -C "strong password"{RESET}"#);
34        anyhow::bail!("Private key file not found.");
35    }
36    println!("  {YELLOW}This function will convert your human password into a digital form hopefully harder to guess. {RESET}");
37    println!();
38    println!("{BLUE}Enter the human easy password to convert:{RESET}");
39    let secret_human_password = secrecy::SecretString::from(
40        inquire::Password::new("")
41            .without_confirmation()
42            .with_display_mode(inquire::PasswordDisplayMode::Masked)
43            .prompt()
44            .log()?,
45    );
46    let secret_first_human_hash_32bytes: [u8; 32] = rsa::sha2::Sha256::digest(secret_human_password.expose_secret().as_bytes()).into();
47    // first try to use the private key from ssh-agent, else use the private file with user interaction
48    let secret_passcode_32bytes: SecretBox<[u8; 32]> =
49        ende::sign_seed_with_ssh_agent_or_private_key_file(&private_key_path_struct, secret_first_human_hash_32bytes).log()?;
50    // hash one more time because signature with private key can be decrypted with the public key
51    let secret_final_human_hash_32bytes: [u8; 32] = rsa::sha2::Sha256::digest(secret_passcode_32bytes.expose_secret()).into();
52    // encode into string that has ascii uppercase, lowercase, numbers and special characters: !, @, $, %, ^, &, *, +, #
53    // This lookup table is missing some letters and numbers to make it non-standard
54    const LOOKUP_TABLE: [char; 64] = [
55        'A', 'B', 'C', 'D', 'E', 'F', 'G', 'H', 'J', 'K', 'L', 'M', 'N', 'P', 'Q', 'R', 'S', 'T', 'U', 'V', 'W', 'Y', 'Z', 'a', 'b', 'c',
56        'd', 'e', 'f', 'g', 'h', 'i', 'j', 'k', 'm', 'n', 'o', 'p', 'q', 'r', 's', 't', 'u', 'w', 'y', 'z', '1', '2', '3', '4', '5', '6',
57        '7', '8', '9', '!', '@', '$', '%', '^', '&', '*', '+', '#',
58    ];
59
60    let mut vec_char: Vec<char> = Vec::new();
61    for one_byte in secret_final_human_hash_32bytes.iter() {
62        let index: usize = (one_byte % 64).into();
63        vec_char.push(LOOKUP_TABLE[index]);
64    }
65    let strong_password = vec_char.into_iter().collect();
66
67    // Strong passwords must use the clipboard. The risk is that it can stay in the clipboard and can be read from the clipboard.
68    // But the same problem is when using a password manager.
69    Ok(strong_password)
70}
