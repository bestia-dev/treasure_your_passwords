12/// List token names from vault.
13///
14/// No need to decrypt the tokens here.
15pub(crate) fn list_tokens_from_vault() -> anyhow::Result<Vec<String>> {
16    let mut ret_vec_string = vec![];
17    println!("  {YELLOW}Check if the encrypted file exists.{RESET}");
18    let private_key_file_name = crate::TREASURE_CONFIG.get().unwrap().treasure_private_key_file_name.to_string();
19
20    let encrypted_path_struct = ende::PathStructInSshFolder::new(format!("{private_key_file_name}.enc"))?;
21    if !encrypted_path_struct.exists() {
22        println!("  {YELLOW}Encrypted file {encrypted_path_struct} does not exist.{RESET}");
23        println!("  {YELLOW}Create the vault and store a secret using the store command. {RESET}");
24        anyhow::bail!("Encrypted file not found.");
25    }
26    println!("  {YELLOW}Open and read the encrypted file.{RESET}");
27    let encrypted_text_with_metadata: String = ende::open_file_b64_get_string(encrypted_path_struct.get_full_file_path())?;
28    // parse json
29    let vec_encrypted_text_with_metadata: Vec<ende::EncryptedTextWithMetadata> = serde_json::from_str(&encrypted_text_with_metadata)?;
30    println!("  {YELLOW}Decrypt the file with ssh-agent or private key.{RESET}");
31    for encrypted_text_with_metadata in vec_encrypted_text_with_metadata.iter() {
32        let iter_token_name = if let Some(iter_token_name) = encrypted_text_with_metadata.token_name.as_ref() {
33            iter_token_name
34        } else {
35            continue;
36        };
37        // here we have the token_name
38        ret_vec_string.push(iter_token_name.to_owned());
39    }
40    Ok(ret_vec_string)
41}

43/// Get secret token from vault.
44///
45/// If exists, decrypt it from file.  
46pub(crate) fn show_secret_token_from_vault(token_name: &str) -> anyhow::Result<SecretString> {
47    println!("  {YELLOW}Check if the ssh private key exists.{RESET}");
48    let private_key_file_name = crate::TREASURE_CONFIG.get().unwrap().treasure_private_key_file_name.to_string();
49
50    println!("  {YELLOW}Check if the encrypted file exists.{RESET}");
51    let encrypted_path_struct = ende::PathStructInSshFolder::new(format!("{private_key_file_name}.enc"))?;
52    if !encrypted_path_struct.exists() {
53        println!("  {YELLOW}Encrypted file {encrypted_path_struct} does not exist.{RESET}");
54        println!("  {YELLOW}Create the vault and store a secret using the store command. {RESET}");
55        anyhow::bail!("Encrypted file not found.");
56    }
57
58    println!("  {YELLOW}Open and read the encrypted file.{RESET}");
59    let encrypted_text_with_metadata: String = ende::open_file_b64_get_string(encrypted_path_struct.get_full_file_path())?;
60    // parse json
61    let vec_encrypted_text_with_metadata: Vec<ende::EncryptedTextWithMetadata> = serde_json::from_str(&encrypted_text_with_metadata)?;
62    println!("  {YELLOW}Decrypt the file with ssh-agent or private key.{RESET}");
63    for encrypted_text_with_metadata in vec_encrypted_text_with_metadata.iter() {
64        let iter_token_name = if let Some(iter_token_name) = encrypted_text_with_metadata.token_name.as_ref() {
65            iter_token_name
66        } else {
67            continue;
68        };
69        if iter_token_name != token_name {
70            continue;
71        }
72        // here we are sure that this is the equal token_name
73        let plain_seed_bytes_32bytes = ende::decode64_from_string_to_32bytes(&encrypted_text_with_metadata.plain_seed_string)?;
74        // use the private key from the encrypted file
75        let private_key_path_struct = ende::PathStructInSshFolder::new(encrypted_text_with_metadata.private_key_file_name.clone())?;
76        if !private_key_path_struct.exists() {
77            eprintln!("{RED}Error: Private key {private_key_path_struct} does not exist. Cannot decrypt.{RESET}");
78            anyhow::bail!("Private key file not found.");
79        }
80
81        let secret_passcode_32bytes: SecretBox<[u8; 32]> =
82            ende::sign_seed_with_ssh_agent_or_private_key_file(&private_key_path_struct, plain_seed_bytes_32bytes)?;
83
84        // decrypt the secret access token string
85        let secret_access_token: SecretString =
86            ende::decrypt_symmetric(secret_passcode_32bytes, encrypted_text_with_metadata.plain_encrypted_text.clone())?;
87        return Ok(secret_access_token);
88    }
89    anyhow::bail!("Token with this name not found.");
90}

92/// Store secret token to vault.
93///
94/// If exists, decrypt it from file.  
95pub(crate) fn store_secret_token_to_vault(token_name: &str) -> anyhow::Result<()> {
96    let mut vec_encrypted_text_with_metadata: Vec<ende::EncryptedTextWithMetadata> = vec![];
97    println!("  {YELLOW}Check if the ssh private key exists.{RESET}");
98    let private_key_file_name = crate::TREASURE_CONFIG.get().unwrap().treasure_private_key_file_name.to_string();
99
100    let private_key_path_struct = ende::PathStructInSshFolder::new(private_key_file_name.clone())?;
101    if !private_key_path_struct.exists() {
102        eprintln!("{RED}Error: Private key {private_key_path_struct} does not exist.{RESET}");
103        println!("  {YELLOW}Create the private key in bash terminal:{RESET}");
104        println!(r#"{GREEN}ssh-keygen -t ed25519 -f {private_key_path_struct} -C "vault for secret tokens"{RESET}"#);
105        anyhow::bail!("Private key file not found.");
106    }
107
108    println!("  {YELLOW}Check if the encrypted file exists.{RESET}");
109    let encrypted_path_struct = ende::PathStructInSshFolder::new(format!("{private_key_file_name}.enc"))?;
110    if encrypted_path_struct.exists() {
111        println!("  {YELLOW}Open and read the encrypted file.{RESET}");
112        let encrypted_text_with_metadata: String = ende::open_file_b64_get_string(encrypted_path_struct.get_full_file_path())?;
113        // parse json
114        vec_encrypted_text_with_metadata = serde_json::from_str(&encrypted_text_with_metadata)?;
115    }
116    println!();
117    println!("{BLUE}Enter the secret token to encrypt:{RESET}");
118    let secret_access_token = secrecy::SecretString::from(
119        inquire::Password::new("")
120            .without_confirmation()
121            .with_display_mode(inquire::PasswordDisplayMode::Masked)
122            .prompt()?,
123    );
124
125    // prepare the random bytes, sign it with the private key, that is the true passcode used to encrypt the secret
126    let (plain_seed_bytes_32bytes, plain_seed_string) = ende::random_seed_32bytes_and_string()?;
127    // first try to use the private key from ssh-agent, else use the private file with user interaction
128    let secret_passcode_32bytes: SecretBox<[u8; 32]> =
129        ende::sign_seed_with_ssh_agent_or_private_key_file(&private_key_path_struct, plain_seed_bytes_32bytes)?;
130    let plain_encrypted_text = ende::encrypt_symmetric(secret_passcode_32bytes, secret_access_token)?;
131
132    // delete the token before writing it with the same token_name
133    if !vec_encrypted_text_with_metadata.is_empty() && delete_token_from_vault(token_name).is_ok() {
134        let encrypted_text_with_metadata: String = ende::open_file_b64_get_string(encrypted_path_struct.get_full_file_path())?;
135        // parse json
136        vec_encrypted_text_with_metadata = serde_json::from_str(&encrypted_text_with_metadata)?;
137    }
138
139    // prepare a struct to save as encoded string
140    let encrypted_text_with_metadata = ende::EncryptedTextWithMetadata {
141        private_key_file_name: private_key_path_struct.get_file_name().to_string(),
142        plain_seed_string,
143        plain_encrypted_text,
144        access_token_expiration: None,
145        refresh_token_expiration: None,
146        token_name: Some(token_name.to_owned()),
147    };
148    vec_encrypted_text_with_metadata.push(encrypted_text_with_metadata);
149
150    let file_text = serde_json::to_string_pretty(&vec_encrypted_text_with_metadata)?;
151    // encode it just to obscure it a little bit
152    let file_text = ende::encode64_from_string_to_string(&file_text);
153
154    std::fs::write(encrypted_path_struct.get_full_file_path(), file_text)?;
155    println!("  {YELLOW}Encrypted text saved to file.{RESET}");
156    Ok(())
157}

159/// Delete secret token from vault.
160///
161/// If exists, delete it from file.  
162pub(crate) fn delete_token_from_vault(token_name: &str) -> anyhow::Result<()> {
163    println!("  {YELLOW}Check if the ssh private key exists.{RESET}");
164    let private_key_file_name = crate::TREASURE_CONFIG.get().unwrap().treasure_private_key_file_name.to_string();
165
166    let private_key_path_struct = ende::PathStructInSshFolder::new(private_key_file_name.clone())?;
167    if !private_key_path_struct.exists() {
168        eprintln!("{RED}Error: Private key {private_key_path_struct} does not exist.{RESET}");
169        println!("  {YELLOW}Create the private key in bash terminal:{RESET}");
170        println!(r#"{GREEN}ssh-keygen -t ed25519 -f {private_key_path_struct} -C "vault for secret tokens"{RESET}"#);
171        anyhow::bail!("Private key file not found.");
172    }
173
174    println!("  {YELLOW}Check if the encrypted file exists.{RESET}");
175    let encrypted_path_struct = ende::PathStructInSshFolder::new(format!("{private_key_file_name}.enc"))?;
176    if !encrypted_path_struct.exists() {
177        println!("  {YELLOW}Encrypted file {encrypted_path_struct} does not exist.{RESET}");
178        println!("  {YELLOW}Create the vault and store a secret using the store command. {RESET}");
179        anyhow::bail!("Encrypted file not found.");
180    }
181
182    println!("  {YELLOW}Open and read the encrypted file.{RESET}");
183    let encrypted_text_with_metadata: String = ende::open_file_b64_get_string(encrypted_path_struct.get_full_file_path())?;
184    // parse json
185    let mut vec_encrypted_text_with_metadata: Vec<ende::EncryptedTextWithMetadata> = serde_json::from_str(&encrypted_text_with_metadata)?;
186    println!("  {YELLOW}Decrypt the file with ssh-agent or private key.{RESET}");
187    let mut index_to_delete = None;
188
189    for (index, encrypted_text_with_metadata) in vec_encrypted_text_with_metadata.iter().enumerate() {
190        let iter_token_name = if let Some(iter_token_name) = encrypted_text_with_metadata.token_name.as_ref() {
191            iter_token_name
192        } else {
193            continue;
194        };
195        if iter_token_name != token_name {
196            continue;
197        }
198        index_to_delete = Some(index);
199        break;
200    }
201    if let Some(index_to_delete) = index_to_delete {
202        println!("  {YELLOW}Delete token {token_name} from vault. {RESET}");
203
204        vec_encrypted_text_with_metadata.remove(index_to_delete);
205
206        let file_text = serde_json::to_string_pretty(&vec_encrypted_text_with_metadata)?;
207        // encode it just to obscure it a little bit
208        let file_text = ende::encode64_from_string_to_string(&file_text);
209
210        std::fs::write(encrypted_path_struct.get_full_file_path(), file_text)?;
211        println!("  {YELLOW}Encrypted text saved to file.{RESET}");
212    } else {
213        anyhow::bail!("Token with this name not found.");
214    }
215
216    Ok(())
