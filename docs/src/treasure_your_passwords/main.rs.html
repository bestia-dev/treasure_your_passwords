26//! ## create the SSH key
27//!
28//! Create the SSH key and protect it with a passcode.
29//!
30//! ```bash
31//! ssh-keygen -t ed25519 -f vault_ssh_1 -C "vault for secret tokens"
32//! ```
33//!
34//! Save the file `treasure_config.json` with the content:
35//!
36//! ```json
37//! {
38//! "treasure_private_key_file_name":"vault_ssh_1"
39//! }
40//! ```
41//!
42//! The program `treasure` will read this file to find the SSH private key in the `~/.ssh` folder.
43//!
44//! ## Use SSH private key to store passwords
45//!
46//! With one SSH private key, we can store many secret tokens.
47//!
48//! ```bash
49//! treasure list
50//! treasure store token_name
51//! treasure show token_name
52//! treasure delete token_name
53//! ```
54//!
55//! ## convert to strong password
56//!
57//! I'd like to have a CLI where to input a humane easy to memorize password and convert it into a strong password.  
58//!
59//! ```bash
60//! treasure strong
61//! ```
62//!
63//! Then sign it with a private key (this encryption is reversible using the public key).  
64//! Then hash it (this is a one way encryption, so nobody can come back to the first secret).  
65//! Finally, convert it into a string long 32 characters with ascii7 characters (lowercase, uppercase, numeric and special characters).  
66//! What makes this conversion secure is: only the user of the private key can convert the easy password into the same strong_password.
67//!
68//! Strong passwords must use the clipboard. The risk is that it can stay in the clipboard and can be read from the clipboard.

133/// The main() function returns ExitCode.
134fn main() -> std::process::ExitCode {
135    match main_returns_anyhow_result() {
136        Err(err) => {
137            eprintln!("{}", err);
138            // eprintln!("Exit program with failure exit code 1");
139            std::process::ExitCode::FAILURE
140        }
141        Ok(()) => std::process::ExitCode::SUCCESS,
142    }
143}
144
145/// The main_returns_anyhow_result() function returns anyhow::Result.
146fn main_returns_anyhow_result() -> anyhow::Result<()> {
147    tracing_init()?;
148    treasure_config_initialize();
149    // super simple argument parsing. There are crates that can parse more complex arguments.
150    match std::env::args().nth(1).as_deref() {
151        None | Some("--help") | Some("-h") => print_help(),
152        Some("strong") => convert_to_strong_password(),
153        Some("list") => list_token_names(),
154        Some("store") => match std::env::args().nth(2).as_deref() {
155            // second argument
156            Some(token_name) => {
157                store_token(token_name);
158            }
159            None => eprintln!("{RED}Error: Missing arguments `token_name`.{RESET}"),
160        },
161        Some("show") => match std::env::args().nth(2).as_deref() {
162            // second argument
163            Some(token_name) => {
164                show_token(token_name);
165            }
166            None => eprintln!("{RED}Error: Missing arguments `token_name`.{RESET}"),
167        },
168        Some("delete") => match std::env::args().nth(2).as_deref() {
169            // second argument
170            Some(token_name) => {
171                delete_token(token_name);
172            }
173            None => eprintln!("{RED}Error: Missing arguments `token_name`.{RESET}"),
174        },
175        _ => eprintln!("{RED}Error: Unrecognized arguments. Try `treasure --help`{RESET}"),
176    }
177    Ok(())
178}

181/// Initialize tracing to file logs/treasure_your_passwords.log.  \
182///
183/// The folder logs/ is in .gitignore and will not be committed.  
184pub fn tracing_init() -> anyhow::Result<()> {
185    let offset = time::UtcOffset::current_local_offset()?;
186    let timer = tracing_subscriber::fmt::time::OffsetTime::new(
187        offset,
188        time::macros::format_description!("[hour]:[minute]:[second].[subsecond digits:6]"),
189    );
190
191    // A filter consists of one or more comma-separated directives
192    // target[span{field=value}]=level
193    // Levels order: 1. ERROR, 2. WARN, 3. INFO, 4. DEBUG, 5. TRACE
194    // ERROR level is always logged.
195    // Add filters to TREASURE_YOUR_PASSWORDS_LOG environment variable for a single execution:
196    // ```bash
197    // TREASURE_YOUR_PASSWORDS_LOG="debug,hyper_util=info,reqwest=info" ./{package_name}
198    // ```
199    let filter = tracing_subscriber::EnvFilter::from_env("TREASURE_YOUR_PASSWORDS_LOG");
200
201    let builder = tracing_subscriber::fmt()
202        .with_file(true)
203        .with_timer(timer)
204        .with_line_number(true)
205        .with_ansi(false)
206        .with_env_filter(filter);
207    if std::env::var("TREASURE_YOUR_PASSWORDS_LOG").is_ok() {
208        // if TREASURE_YOUR_PASSWORDS_LOG exists than enable tracing to file
209        let file_appender = tracing_appender::rolling::RollingFileAppender::builder()
210            .rotation(tracing_appender::rolling::Rotation::DAILY)
211            .filename_prefix("treasure_your_passwords")
212            .filename_suffix("log")
213            .build("logs")
214            .expect("initializing rolling file appender failed");
215        builder.with_writer(file_appender).init();
216    } else {
217        builder.init();
218    };
219
220    Ok(())
221}

223/// Trait to log the error from Result before propagation with ?.
224pub trait ResultLogError<T, E>: Sized {
225    fn log(self) -> Self;
226}
227
228/// Implements LogError for anyhow::Result.
229impl<T, E: std::fmt::Debug> ResultLogError<T, E> for core::result::Result<T, E> {
230    #[inline(always)]
231    #[track_caller]
232    fn log(self) -> Self {
233        self.inspect_err(|err| tracing::error!(?err))
234    }
235}

239/// Application state (static) is initialized only once in the main() function.
240///
241/// And then is accessible all over the code.
242fn treasure_config_initialize() {
243    if TREASURE_CONFIG.get().is_some() {
244        return;
245    }
246
247    let treasure_config_json = std::fs::read_to_string("treasure_config.json").unwrap();
248    let treasure_config: TreasureConfig = serde_json::from_str(&treasure_config_json).unwrap();
249    let _ = TREASURE_CONFIG.set(treasure_config);
250}
